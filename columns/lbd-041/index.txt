先说  push  API，直接将元素加⼊队列，同时记录队尾元素，因为队尾元素
相当于栈顶元素，如果要  top  查看栈顶元素的话可以直接返回：
class MyStack {

    int top_elem = 0;
    /** 添加元素到栈顶 */
    public void push(int x) {
        // x 是队列的队尾，是栈的栈顶
        q.offer(x);
        top_elem = x;
    }
    /** 返回栈顶元素 */
    public int top() {
        return top_elem;
    }
}
