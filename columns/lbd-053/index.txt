# intervals 形如 [[1,3],[2,6]...]
def merge(intervals):
    if not intervals: return []
    # 按区间的 start 升序排列
    intervals.sort(key=lambda intv: intv[0])
    res = []
    res.append(intervals[0])
    for i in range(1, len(intervals)):
        curr = intervals[i]
        # res 中最后⼀个元素的引⽤
        last = res[-1]
        if curr[0] <= last[1]:
            # 找到最⼤的 end
            last[1] = max(last[1], curr[1])
        else:
            # 处理下⼀个待合并区间
            res.append(curr)
    return res
